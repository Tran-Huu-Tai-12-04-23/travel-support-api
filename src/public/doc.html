3. Create User

Endpoint: POST /api/users

Input:

{
    "username": "exampleUsername",
    "password": "examplePassword",
    "userDetail": {
        "name": "John Doe",
        "fullName": "John A. Doe",
        "address": "123 Main St",
        "email": "john@example.com",
        "phoneNumber": "123-456-7890"
    }
}

Output:

{
    "_id": "generatedUserId",
    "username": "exampleUsername",
    "login_history": "timestamp",
    "userDetail": {
        "name": "John Doe",
        "fullName": "John A. Doe",
        "address": "123 Main St",
        "email": "john@example.com",
        "phoneNumber": "123-456-7890"
    },
    "createdAt": "timestamp",
    "updatedAt": "timestamp"
}
